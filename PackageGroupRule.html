def generate_clauses(repo)
  ids = parameters[:units].collect do |unit|
    #{'name' => {"$regex" => unit[:name]}}
    unless unit[:name].blank?
      PackageGroup.search(unit[:name], 0, 0, [repo.pulp_id]).collect(&:package_group_id)
    end
  end.compact.flatten
  {"id" => {"$in" => ids}}
end
