def memory
  User.as :admin do
    objs = Hash.new(0)
    ObjectSpace.each_object do |o|
      objs[o.class] += 1
    end
    output = objs.sort_by { |c, n| n }.last(30)
    render :text => PP.pp(output, "")
  end
end
