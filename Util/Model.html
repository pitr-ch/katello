def self.labelize(name)
  if name
    name.ascii_only? ? name.gsub(/[^a-z0-9\-_]/i,"_") : uuid
  end
end

def self.table_to_class name
  class_name = @@table_to_model_hash[name] || name.classify
  class_name.constantize
rescue NameError => e
  # constantize throws NameError
  return nil
end
